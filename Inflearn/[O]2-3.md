                    if (a[i] === b[i]) {

                    } else if (a[i] === b[i] + 1) {
                        console.log("A");

                        console.log("A");
                    } else {
                        console.log("B");
                    }
                }
            }

            let b = [1, 1, 2, 2, 3];
            console.log(solution(a, b));
